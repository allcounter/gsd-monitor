**2. SidebarView.swift** (sidebar with empty state):
```swift
import SwiftUI

struct SidebarView: View {
    let projects: [Project]
    @Binding var selectedProjectID: UUID?

    var body: some View {
        if projects.isEmpty {
            ContentUnavailableView(
                "No Projects Found",
                systemImage: "folder.badge.questionmark",
                description: Text("GSD projects will appear here when discovered in ~/Developer")
            )
        } else {
            List(projects, selection: $selectedProjectID) { project in
                NavigationLink(value: project.id) {
                    Text(project.name)
                }
            }
            .navigationTitle("Projects")
        }
    }
}

#Preview("Empty State") {
    SidebarView(projects: [], selectedProjectID: .constant(nil))
}

#Preview("With Projects") {
    SidebarView(
        projects: PreviewData.projects,
        selectedProjectID: .constant(PreviewData.projects.first?.id)
    )
}
```

import SwiftUI

struct DetailView: View {
    let selectedProject: Project?

    var body: some View {
        if let project = selectedProject {
            // Phase 1: Simple text, Phase 4 will build full roadmap view
            VStack(spacing: 16) {
                Text(project.name)
                    .font(.largeTitle)
                    .fontWeight(.bold)

                Text("Roadmap view will appear here")
                    .foregroundStyle(.secondary)
            }
            .frame(maxWidth: .infinity, maxHeight: .infinity)
        } else {
            ContentUnavailableView(
                "Select a Project",
                systemImage: "sidebar.left",
                description: Text("Choose a project from the sidebar to view its roadmap")
            )
        }
    }
}

**Key architectural decisions (from RESEARCH.md):**
- Use **two-column** NavigationSplitView (sidebar + detail), NOT three-column
- Use `.navigationSplitViewStyle(.balanced)` for proper macOS column widths
- Use `@State private var selectedProjectID: UUID?` (value type, not reference type) - avoids @Observable memory leak pitfall from PITFALLS.md
- Use `ContentUnavailableView` (macOS 14+) for empty states - native macOS design
- Test BOTH light and dark mode in previews (automatic theme support verification)

**What to avoid:**
- Do NOT use `@State` with `@Observable` reference types (memory leak risk per PITFALLS.md Pitfall 2)
- Do NOT create a ViewModel yet (Phase 1 is static UI skeleton only)
- Do NOT use three-column NavigationSplitView (overengineering per RESEARCH.md Pitfall 3)
- Do NOT implement actual file scanning yet (that's Phase 2)
- Do NOT use LazyVStack (List is better for macOS per PITFALLS.md Pitfall 4)
