   ```swift
       StatCardView(icon: "percent", value: "\(completionPercent)%", label: "Complete", accentColor: Theme.brightGreen)

   }
   ```

**Computed properties:**

- `totalPhases`: `project.roadmap?.phases.count ?? 0`

- `completionPercent`: REUSE the same logic as `DetailView.overallProgress(for:)` to avoid divergence. Copy the exact calculation pattern:
  ```swift
  private var completionPercent: Int {
      guard let roadmap = project.roadmap, !roadmap.phases.isEmpty else { return 0 }
      let plans = project.plans ?? []
      let phaseContributions = roadmap.phases.map { phase -> Double in
          if phase.status == .done { return 1.0 }
          let phasePlans = plans.filter { $0.phaseNumber == phase.number }
          guard !phasePlans.isEmpty else { return 0.0 }
          let done = phasePlans.filter { $0.status == .done }.count
          return Double(done) / Double(phasePlans.count)
      }
      return Int((phaseContributions.reduce(0, +) / Double(roadmap.phases.count)) * 100)
  }
  ```

- `activePhases`: `project.roadmap?.phases.filter { $0.status == .inProgress }.count ?? 0`

- `executionTime`: `project.state?.totalExecutionTime ?? "—"` (em dash for projects without data)

**Only show StatsGridView when roadmap has phases** — wrap the entire view body in a guard or conditional. If no roadmap, return EmptyView.
